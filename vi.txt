vi test1.txt # создать новый документ

ESC      # войти в командный режим
Ctrl + F # Пролистать файл вниз
Ctrl + B # Пролистать файл вверх

i   # начать ввод текста с позиции, где находится курсор, сдвигая текст
a   # начать ввод текста после позиции курсора
o   # начать ввод текста с новой строки
O   # вставить пустую строку над текущей и перейти в режим редактирования
v   # перейти в режив выделения текста
y   # скопировать выделенный текст
yy  # скопировать строку
p   # вставить текст из буфера обмена
x   # удаляет символ, на котором курсор
dd  # удалить строку
dw  # удалить слово (с позиции курсора до первого пробела)
u   # отменить последнюю операцию
.   # повторить последнюю операцию
R   # заменяет текст, новым (как будто нажали клавишу insert)
r   # заменяет 1 символ

h   # перемещает курсор влево
j   # перемещает курсор вниз
k   # перемещает курсор вверх
l   # перемещает курсор вправо
w   # перемещает курсор на начало нового слова
b   # перемещает курсор на начало предыдущего слова
0   # перемещает курсор в начало строки
$   # перемещает курсор в конец строки
G   # перемещает курсор на последнюю строку файла
10G # перемещает курсор на 10 строку файла
1G  # перемещает курсор на 1 строку

:r test2.txt # вставить содержимое test2.txt начиная с позиции курсора
:q           # выход
:q!          # выход без сохранения
:wq          # выход с сохранением
:w           # сохранить файл
