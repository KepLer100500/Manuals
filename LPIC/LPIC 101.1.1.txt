HAL  # предоставляет абстракцию железа ОС (устарела)
Dbus # предоставляет информацию о железе (устарела)
udev # современный менеджер устройств

Devfs - (устарела) виртуальная файловая система (монтируется в памяти), функционал перешёл в udev. Монтируется в /dev

mem     # образ физической памяти
null    # пустое устройство
pts/    # псевдотерминалы
urandom # генератор случайных чисел
sdx     # блочные устройства (жёсткие диски, usb накопители)
tty     # аппаратные терминалы

qwerty > /dev/null 2>&1       # Отправить в мусор все выводимые в терминал сообщения
head -1 /dev/urandom > ~/test # Записать случайный набор символов в ~/test

Sysfs - (устарела) виртуальная файловая система. Экспортирует информацию о драйверах и устройствах на уровень пользователя. Монтируется в /sys

devices/ # все устройства, зарегестрированные в ядре
bus/     # перечень шин
drivers/ # драйверы
block/   # каталог блочных устройств
class/   # группировка устройств по классам

ls -lh /sys/bus/usb/devices/ # информация об устройствах usb
ls -lh /sys/block/           # информация о блочных устройствах
ls /sys/class/net/           # информация о сетевых интерфейсах
ip addr                      # информация о сетевых интерфейсах

Procfs - иерархическое представление всех процессов в системе. Монтируется в /proc

PID/     # информация о конкретном процессе
cpuinfo/ # сведения о ЦПУ
devices/ # перечень настроенных устройств
mounts/  # смонтированные файловые системы
sys/     # доступная для редактирования информация о системе

cat /proc/cpuinfo # информация о процессоре
cat /proc/mounts  # информация о всех подмонтированных файловых системах
cat /proc/sys/fs/file-max         # максимально возможное количество одновременно откртых файлов (можно изменить до перезагрузки(налету))
cat /proc/sys/net/ipv4/ip_forward # включен ли сейчас форвардинг между сетевыми картами (можно изменить до перезагрузки(налету))
