#Вывод списка оборудования

lsmod    # информация о модулях ядра
lspci    # информация об устройствах PCI
lspcmcia # информация об устройствах PCMCIA (для ноутбуков)
lsusb    # информация о шине USB
lshal    # база устройств HAL (устарело)
lshw     # детальная информация о комплектующих (устарело)

udevadm

info    # получение информации из БД
trigger # запросить события для устройства
settle  # дождаться завершения обработки
control # управление демоном
monitor # следить за событиями
test    # симулировать запуск события

sudo yum install usbutils # установка утилит для работы с USB

udevadm info -a -p /sys/class/net/wlan0 # показать инфу о сетевой карте
udevadm monitor --property --kernel # начать мониторинг событий по ядру (включение/отключение устройств увидим)
udevadm test /sys/block/mmcblk0 # протестить события, происходящие на sd карте

Управление модулями ядра

lsmod    # информация о модулях ядра
modinfo  # информация о конкретном модуле
rmmod    # удаление модуля ядра
insmod   # установка модуля ядра
modprobe # деликатное удаление или добавление модуля ядра

modinfo sg  # показать информацию о модуле sg
rmmod sg    # удалить модуль sg
modprobe sg # установить модуль sg
sestatus    # статус работы SELinux (система принудительного контроля доступа)
