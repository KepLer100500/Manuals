# Управление библиотеками в Linux

ls /lib             # тут библиотки, которые используются программами из /bin
ls /usr/lib         # тут библиотки, которые используются программами из /usr/bin
cat /etc/ld.so.conf # обозначает пути, где могут лежать дополнительные библиотеки
ldd /bin/ls         # показать, какими библиотеками пользуется программа ls

export LD_LIBRARY_PATH=/opt/some_soft/lib # Добавить в переменную окружения LD_LIBRARY_PATH новый путь для библиотек

ldconfig # обновать кеш загруженых в память библиотек
