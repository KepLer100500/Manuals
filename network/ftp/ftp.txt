ftp 192.168.0.1         # соединиться с ftp сервером
ftp ftpuser@192.168.0.1 # соединиться с ftp сервером

pwd            # покажет текущую директорию на ftp-сервере
ls             # покажет список файлов и каталогов в текущей директории
cd [folder]    # для перехода в нужную директорию
mkdir [folder] # создание новой директории (каталога)
rmdir [folder] # удаление директории (каталога)
delete [file]  # удаляет файл на удаленном ftp сервере

get file.zip             # скачать файл на локальную машину
lcd /home/user1          # изменить директорию на локальной машине куда будут скачиваться файлы (текущий каталог)
mget test1.sql test2.sql # скачать несколько файлов
mget *.sql               # скачать несколько файлов

put test1.sql            # команда для загрузки одного файла на ftp сервер
mput test1.sql test2.sql # для загрузки нескольких файлов сразу

tick # команда будет отображать счетчик байтов:
hash # команда после которой ftp будет печатать символ «#» каждые 1024 байт данных:
