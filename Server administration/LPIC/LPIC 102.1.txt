Следующие папки могут быть смонтированы как отдельные разделы жёсткого диска

/     # корневая файловая система, самый большой раздел 
/boot # загрузочный раздел (тут хранится ядро, ~Мб 100)
/home # домашние папки
/root # домашняя папка суперпользователя
/etc  # конфигурация системы и её компонентов
/opt  # папка для ПО от третьих поставщиков
/var  # часто меняющиеся данные (логи, программые данные, имеет смысл вынести на SSD)
/usr  # все установленные пакеты программ, документация, исходники ядра
/tmp  # временные файлы
swap  # раздел подкачки, никуда не монтируется

fdisk # утилита для разбиения жесткого диска
mkfs  # утилита создания файловой системы

sda1-sda4   # зарезервированы для основных разделов (разделы бывают основные и логические)
sda5-sda[x] # тут начинаются логические разделы

sudo fdisk -l       # показывает диски и разделы
sudo fdisk /dev/sdb # начать работать с новым жёстким диском
m       # справка по всем командам
n       # созадть раздел
-> p    # primary (первичный раздел) (p/l)
-> 1    # пусть будет первым по счёту (по умолчанию 1)
-> 2048 # откуда будет начинаться первый сектор раздела (по умолчанию 2048)
-> +2GB # этот раздел будет в 2 гигабайта
w       # записать изменения
t       # для изменения id раздела (L - вывести список всех id, a5 - FreeBSD, 82 - Linux swap)
p       # показать в табличном виде все созданные разделы
d       # удалить раздел

mkfs.ext4 /dev/sdb5 # форматировать раздел sdb5 как ext4

/etc/fstab # конфигурационный файл, отвечающий за монтирование файловых систем
blkid      # показать что за устройство на каком UUID

swapon и swapoff # команды работы с разделом подкачки
mkswap           # утилита создания раздела или файла подкачки
swapon -s        # показать текущие файлы подкачки

mkdir /mounted 
mount /dev/sdb5 /mounted # подмонтировать отформатированный ранее раздел sdb5  
umount /mounted/         # отмонтировать sdb5

LVM - Linux Volume Manager - система управления томами для линукс

pvcreate /dev/sdb1                      # создать физический том
vgcreate virt_group /dev/sdb1 /dev/sdb2 # создать виртуальную группу томов из sdb1 и sdb2
lvcreate -n lv1 -L 2G virt_group        # создать логический том lv1 (-n - новый, -L - размер)  

pvdisplay # показывает физические тома
vgdisplay # показать виртуальные группы томов
lvdisplay # показать логические тома

fdisk -l # посмотрим на новые созданные блочные устройства

mkfs.ext4 /dev/virt_group/lv1 # теперь можно отформатировать созданные логические тома

lvresize -L 5GB virt_group/lv2 # изменть размер второго логического тома (надо ещё файловую систему resize!)
