### Вход / выход
en / enable            # вход в привилегированный режим
disable / exit         # выход из привилегированного режима
write memory / wr mem  # сохранить конфигурацию

### Показать конфиги
show running-config         # конфиг устройства
do show run                 # показать конфиг из любого режима

### Задать пароль на привелегированный режим
enable password cisco       # задать пароль - "cisco"
enable secret cisco2        # задать шифрованный пароль (имеет приоритет)

### Создать пользователя
configure terminal # conf t                # режим глобального конфигурирования
username admin privilege 15 password cisco # создать пользователя admin с максимальными привелегиями и паролем cisco
service password-encryption                # зашифровать пароль в конфиге
line console 0                             # режим конфигурирования терминальных линий
login local                                # используем локальную базу авторизации
end                                        # выйти сразу из всех режимов конфигурации

### Найстройка ip адреса устройства (конфигурировать надо логические, а не физические интерфейсы)
interface vlan1                        # режим конфигурирования интерфейсов 
ip address 192.168.1.100 255.255.255.0 # задать ip адрес
no shutdown                            # поднять интерфейс

### Выбираем тип удалённого подключения и включаем для него авторизацию
line vty 0 4           # настройка виртуальной терминальной линии
transport input telnet # разрешаем telnet
login local            # используем локальную базу авторизации
